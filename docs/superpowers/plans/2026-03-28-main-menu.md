# Main Menu Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a main menu screen with a random celestial background scene, atmospheric text, and music that plays before entering the game.

**Architecture:** Add a `GamePhase` enum (Menu/Playing) to the App struct. In Menu phase, render the 3D scene (sky + stars + optionally a planet/star) with a slow camera drift, overlay egui menu text. "Continue" transitions to Playing phase (current behavior). Background scene generated from a time-based seed using existing rendering systems.

**Tech Stack:** egui (already used for HUD), existing renderer (sky, stars, planets), existing audio (music).

```rust
// crates/spaceaway/src/menu.rs
//! Main menu: random celestial background, atmospheric text, music.

use sa_render::{Camera, DrawCommand, MeshStore, MeshMarker, Renderer, DriveRenderParams};
use sa_render::planet_mesh;
use sa_universe::{MasterSeed, generate_system, generate_star};
use sa_universe::sector::{generate_sector, SectorCoord};
use sa_math::WorldPos;
use sa_core::Handle;
use glam::{Mat4, Vec3};
use rand::Rng;

const QUOTES: &[&str] = &[
    "The universe is vast and indifferent.",
    "Every star is a sun. Every sun, a world.",
    "In space, no one can hear you wonder.",
    "The cold between stars is patient.",
    "You are small. The universe does not care.",
    "Mystery is the only compass.",
    "The silence out here has weight.",
    "Every light in the sky is a place you could go.",
    "Fear and wonder are the same thing, out here.",
    "The void doesn't end. Neither does curiosity.",
];

/// What celestial object is shown as the menu background.
enum MenuScene {
    /// Galaxy panorama — just sky + stars, camera in galactic disc facing core
    GalaxyPanorama { galactic_pos: WorldPos },
    /// Deep space — far from disc, minimal stars
    DeepSpace { galactic_pos: WorldPos },
    /// Planet close-up — a generated planet mesh at 1.5-2x radius
    Planet {
        galactic_pos: WorldPos,
        mesh: Handle<MeshMarker>,
        offset_m: [f32; 3], // planet position relative to camera in meters
    },
    /// Star close-up — a generated star mesh with corona
    Star {
        galactic_pos: WorldPos,
        mesh: Handle<MeshMarker>,
        offset_m: [f32; 3],
    },
}

pub struct MainMenu {
    scene: MenuScene,
    camera: Camera,
    quote: &'static str,
    drift_yaw: f32,  // slow camera rotation
    time: f32,
}

impl MainMenu {
    /// Generate a random menu scene from a time-based seed.
    pub fn new(mesh_store: &mut MeshStore, device: &wgpu::Device) -> Self {
        let seed = std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH)
            .unwrap_or_default()
            .as_secs();
        let mut rng = sa_universe::Rng64::new(seed);

        // Pick a quote
        let quote_idx = (rng.next_u64() as usize) % QUOTES.len();
        let quote = QUOTES[quote_idx];

        // Pick scene type by weight
        let roll = rng.next_f32();
        let (scene, camera) = if roll < 0.20 {
            // Galaxy panorama (20%)
            Self::make_galaxy_scene(&mut rng)
        } else if roll < 0.30 {
            // Deep space (10%)
            Self::make_deep_space_scene(&mut rng)
        } else if roll < 0.55 {
            // Planet close-up (25%)
            Self::make_planet_scene(&mut rng, mesh_store, device)
        } else if roll < 0.75 {
            // Gas giant (20%) — same as planet but search for gas giant
            Self::make_planet_scene(&mut rng, mesh_store, device)
        } else if roll < 0.85 {
            // Star close-up (10%)
            Self::make_star_scene(&mut rng, mesh_store, device)
        } else {
            // Galaxy panorama fallback (remaining)
            Self::make_galaxy_scene(&mut rng)
        };

        Self {
            scene,
            camera,
            quote,
            drift_yaw: 0.0,
            time: 0.0,
        }
    }

    fn make_galaxy_scene(rng: &mut sa_universe::Rng64) -> (MenuScene, Camera) {
        // Position in the galactic disc, facing toward the center
        let angle = rng.range_f32(0.0, std::f32::consts::TAU);
        let dist = rng.range_f32(15000.0, 30000.0); // ly from center
        let x = dist * angle.cos() as f64;
        let z = dist * angle.sin() as f64;
        let y = rng.range_f64(-200.0, 200.0); // near disc plane
        let pos = WorldPos::new(x, y, z);

        let mut cam = Camera::new();
        cam.position = pos;
        // Face toward galactic center
        cam.yaw = (-(x as f32)).atan2(z as f32);
        cam.pitch = rng.range_f32(-0.1, 0.1);

        (MenuScene::GalaxyPanorama { galactic_pos: pos }, cam)
    }

    fn make_deep_space_scene(rng: &mut sa_universe::Rng64) -> (MenuScene, Camera) {
        // Far above or below the galactic plane
        let x = rng.range_f64(-5000.0, 5000.0);
        let y = rng.range_f64(5000.0, 15000.0) * if rng.next_f32() > 0.5 { 1.0 } else { -1.0 };
        let z = rng.range_f64(-5000.0, 5000.0);
        let pos = WorldPos::new(x, y, z);

        let mut cam = Camera::new();
        cam.position = pos;
        cam.yaw = rng.range_f32(0.0, std::f32::consts::TAU);
        cam.pitch = rng.range_f32(-0.3, 0.3);

        (MenuScene::DeepSpace { galactic_pos: pos }, cam)
    }

    fn make_planet_scene(
        rng: &mut sa_universe::Rng64,
        mesh_store: &mut MeshStore,
        device: &wgpu::Device,
    ) -> (MenuScene, Camera) {
        // Find a star with planets
        let star_seed = rng.next_u64();
        let star = generate_star(star_seed);
        let system = generate_system(&star, star_seed);

        if let Some(planet) = system.planets.first() {
            let radius_m = planet.radius_earth as f64 * 6_371_000.0;
            let mesh = match planet.planet_type {
                sa_universe::PlanetType::Rocky => planet_mesh::build_rocky_planet_mesh(
                    4, radius_m as f32, planet.sub_type, planet.color_seed),
                _ => planet_mesh::build_gas_giant_mesh(
                    4, radius_m as f32, planet.sub_type, planet.color_seed),
            };
            let handle = mesh_store.upload(device, &mesh);

            // Camera at 2x radius, slightly above
            let view_dist = radius_m * 2.0;
            let offset = [view_dist as f32, (radius_m * 0.3) as f32, 0.0];

            let mut cam = Camera::new();
            cam.position = WorldPos::ORIGIN;
            cam.yaw = -std::f32::consts::FRAC_PI_2; // face -X toward planet
            cam.pitch = -0.15;

            (MenuScene::Planet {
                galactic_pos: WorldPos::ORIGIN,
                mesh: handle,
                offset_m: [-offset[0], offset[1], 0.0], // planet is in -X
            }, cam)
        } else {
            Self::make_galaxy_scene(rng)
        }
    }

    fn make_star_scene(
        rng: &mut sa_universe::Rng64,
        mesh_store: &mut MeshStore,
        device: &wgpu::Device,
    ) -> (MenuScene, Camera) {
        let star_seed = rng.next_u64();
        let star = generate_star(star_seed);
        let radius_m = star.radius as f64 * 696_000_000.0;

        let mesh = planet_mesh::build_star_mesh(3, radius_m as f32, star.color, star_seed);
        let handle = mesh_store.upload(device, &mesh);

        let view_dist = radius_m * 3.0;
        let offset = [-(view_dist as f32), (radius_m * 0.2) as f32, 0.0];

        let mut cam = Camera::new();
        cam.position = WorldPos::ORIGIN;
        cam.yaw = -std::f32::consts::FRAC_PI_2;
        cam.pitch = -0.05;

        (MenuScene::Star {
            galactic_pos: WorldPos::ORIGIN,
            mesh: handle,
            offset_m: offset,
        }, cam)
    }

    /// Update camera drift.
    pub fn update(&mut self, dt: f32) {
        self.time += dt;
        self.drift_yaw += 0.002 * dt; // very slow rotation ~0.1°/s
        self.camera.yaw += 0.002 * dt;
    }

    /// Get the camera for rendering.
    pub fn camera(&self) -> &Camera {
        &self.camera
    }

    /// Get the galactic position for sky/star rendering.
    pub fn galactic_position(&self) -> WorldPos {
        match &self.scene {
            MenuScene::GalaxyPanorama { galactic_pos }
            | MenuScene::DeepSpace { galactic_pos }
            | MenuScene::Planet { galactic_pos, .. }
            | MenuScene::Star { galactic_pos, .. } => *galactic_pos,
        }
    }

    /// Get draw commands for any 3D objects in the scene (planet/star meshes).
    pub fn draw_commands(&self) -> Vec<DrawCommand> {
        match &self.scene {
            MenuScene::Planet { mesh, offset_m, .. } => {
                vec![DrawCommand {
                    mesh: *mesh,
                    model_matrix: Mat4::from_translation(Vec3::from(*offset_m)),
                    pre_rebased: true,
                }]
            }
            MenuScene::Star { mesh, offset_m, .. } => {
                vec![DrawCommand {
                    mesh: *mesh,
                    model_matrix: Mat4::from_translation(Vec3::from(*offset_m)),
                    pre_rebased: true,
                }]
            }
            _ => vec![], // sky-only scenes
        }
    }

    /// Render the menu overlay via egui. Returns true if "Continue" was clicked.
    pub fn render_egui(&self, ctx: &egui::Context) -> bool {
        let mut start_game = false;

        // Semi-transparent dark overlay for readability
        egui::Area::new(egui::Id::new("menu_bg"))
            .fixed_pos(egui::pos2(0.0, 0.0))
            .show(ctx, |ui| {
                let screen = ctx.screen_rect();
                ui.painter().rect_filled(
                    screen,
                    0.0,
                    egui::Color32::from_rgba_unmultiplied(0, 0, 5, 80),
                );
            });

        egui::CentralPanel::default()
            .frame(egui::Frame::NONE)
            .show(ctx, |ui| {
                ui.vertical_centered(|ui| {
                    let screen_h = ui.available_height();

                    ui.add_space(screen_h * 0.25);

                    // Title: spaced-out letters
                    ui.label(
                        egui::RichText::new("S  P  A  C  E  A  W  A  Y")
                            .color(egui::Color32::from_rgb(200, 205, 215))
                            .size(42.0)
                            .strong(),
                    );

                    ui.add_space(60.0);

                    // Menu items
                    let btn_style = egui::RichText::new("▸  Continue")
                        .color(egui::Color32::from_rgb(160, 165, 180))
                        .size(22.0);
                    if ui.add(egui::Label::new(btn_style).sense(egui::Sense::click())).clicked() {
                        start_game = true;
                    }

                    ui.add_space(15.0);

                    // Future items (grayed out)
                    ui.label(
                        egui::RichText::new("▸  New Game")
                            .color(egui::Color32::from_rgb(80, 80, 90))
                            .size(22.0),
                    );
                    ui.add_space(15.0);
                    ui.label(
                        egui::RichText::new("▸  Settings")
                            .color(egui::Color32::from_rgb(80, 80, 90))
                            .size(22.0),
                    );
                    ui.add_space(15.0);
                    // Quit not needed — just close the window

                    ui.add_space(60.0);

                    // Atmospheric quote
                    ui.label(
                        egui::RichText::new(self.quote)
                            .color(egui::Color32::from_rgb(100, 105, 120))
                            .size(14.0)
                            .italics(),
                    );
                });
            });

        start_game
    }
}
```

- [ ] **Step 4: Route frame logic through GamePhase**

The main change: wrap the existing game loop (everything in `RedrawRequested` after input handling) in an `if self.phase == GamePhase::Playing { ... }` block. Before it, add the menu rendering.

In the `RedrawRequested` handler:

```rust
// Update frame time
self.time.tick();
let dt = self.time.delta_seconds() as f32;

if self.phase == GamePhase::Menu {
    // --- MENU PHASE ---
    if let Some(menu) = &mut self.menu {
        menu.update(dt);
        self.audio.update(dt);

        // Render
        if let (Some(gpu), Some(renderer)) = (&self.gpu, &mut self.renderer) {
            // Update star field for menu's galactic position
            let menu_pos = menu.galactic_position();
            let needs_rebuild = self.star_streaming.update(menu_pos, dt);
            if needs_rebuild {
                let verts = self.star_streaming.build_vertices(menu_pos);
                renderer.star_field.update_star_buffer(&gpu.device, &verts);
            }

            // Get menu draw commands (planet/star mesh if applicable)
            let commands = menu.draw_commands();
            let screen_draws: Vec<sa_render::ScreenDrawCommand<'_>> = vec![];
            let drive_params = DriveRenderParams::default();

            if let Some(mut frame_ctx) = renderer.render_frame(
                gpu,
                menu.camera(),
                &commands,
                &screen_draws,
                Vec3::new(0.5, -0.8, -0.3),
                menu_pos,
                &drive_params,
            ) {
                // Render menu egui overlay
                if let Some(ui_sys) = &mut self.ui_system {
                    let start_game = ui_sys.render_menu(
                        &gpu.device,
                        &gpu.queue,
                        &mut frame_ctx.encoder,
                        &frame_ctx.view,
                        menu,
                    );
                    if start_game {
                        self.phase = GamePhase::Playing;
                        self.menu = None;
                        // Reload stars for game starting position
                        self.star_streaming.force_load(self.galactic_position);
                        if let (Some(gpu2), Some(renderer2)) = (&self.gpu, &mut self.renderer) {
                            let verts = self.star_streaming.build_vertices(self.galactic_position);
                            renderer2.star_field.update_star_buffer(&gpu2.device, &verts);
                        }
                        self.audio.set_power(true);
                        log::info!("Starting game — entering ship");
                    }
                }

                Renderer::submit_frame(gpu, frame_ctx);
            }
        }
    }
    // Request next frame
    if let Some(window) = &self.window {
        window.request_redraw();
    }
    return; // skip game logic
}

// --- PLAYING PHASE (existing game loop) ---
// ... all existing code ...
```
